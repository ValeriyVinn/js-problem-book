              const filter = function (array, test) {
                const filteredArray = [];
            
                for (const el of array) {
                    const passed = test(el);
            
                    if (passed) {
                        filteredArray.push(el);
                    }
                }
            
                return filteredArray;
            };
            
            const callback1 = function (value) {
                return value >= 3;
            };
            
            const r1 = filter([1, 2, 3, 4, 5], callback1);
